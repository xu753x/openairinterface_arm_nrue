273 static int trx_usrp_start(openair0_device *device) {
274  usrp_state_t *s = (usrp_state_t *)device->priv;
275 
276  // setup GPIO for TDD, GPIO(4) = ATR_RX
277  //set data direction register (DDR) to output
278  s->usrp->set_gpio_attr("FP0", "DDR", 0xfff, 0xfff);
279  //set lower 7 bits to be controlled automatically by ATR (the rest 5 bits are controlled manually)
280  s->usrp->set_gpio_attr("FP0", "CTRL", 0x7f,0xfff);
281  //set pins 4 (RX_TX_Switch) and 6 (Shutdown PA) to 1 when the radio is only receiving (ATR_RX)
282  s->usrp->set_gpio_attr("FP0", "ATR_RX", (1<<4)|(1<<6), 0x7f);
283  // set pin 5 (Shutdown LNA) to 1 when the radio is transmitting and receiveing (ATR_XX)
284  // (we use full duplex here, because our RX is on all the time - this might need to change later)
285  s->usrp->set_gpio_attr("FP0", "ATR_XX", (1<<5), 0x7f);
286  // set the output pins to 1
287  s->usrp->set_gpio_attr("FP0", "OUT", 7<<7, 0xf80);
288 
289  s->wait_for_first_pps = 1;
290  s->rx_count = 0;
291  s->tx_count = 0;
292  //s->first_tx = 1;
293  //s->first_rx = 1;
294  s->rx_timestamp = 0;
295 
296  s->usrp->set_time_next_pps(uhd::time_spec_t(0.0));
297  // wait for the pps to change
298  uhd::time_spec_t time_last_pps = s->usrp->get_time_last_pps();
299  while (time_last_pps == s->usrp->get_time_last_pps()) {
300  boost::this_thread::sleep(boost::posix_time::milliseconds(1));
301  }
302 
303  uhd::stream_cmd_t cmd(uhd::stream_cmd_t::STREAM_MODE_START_CONTINUOUS);
304  cmd.time_spec = uhd::time_spec_t(1.0);
305  cmd.stream_now = false; // start at constant delay
306  s->rx_stream->issue_stream_cmd(cmd);
307 
308  return 0;
309 }

334 static int trx_usrp_write(openair0_device *device,
335  openair0_timestamp timestamp,
336  void **buff,
337  int nsamps,
338  int cc,
339  int flags) {
340  int ret=0;
341  usrp_state_t *s = (usrp_state_t *)device->priv;
342  int nsamps2; // aligned to upper 32 or 16 byte boundary
343 
344  int flags_lsb = flags&0xff;
345  int flags_msb = (flags>>8)&0xff;
346 
347  int end;
348  openair0_thread_t *write_thread = &device->write_thread;
349  openair0_write_package_t *write_package = write_thread->write_package;
350 
351  AssertFatal( MAX_WRITE_THREAD_BUFFER_SIZE >= cc,"Do not support more than %d cc number\n", MAX_WRITE_THREAD_BUFFER_SIZE);
352 
353  boolean_t first_packet_state=false,last_packet_state=false;
354 
355  if (flags_lsb == 2) { // start of burst
356  // s->tx_md.start_of_burst = true;
357  // s->tx_md.end_of_burst = false;
358  first_packet_state = true;
359  last_packet_state = false;
360  } else if (flags_lsb == 3) { // end of burst
361  //s->tx_md.start_of_burst = false;
362  //s->tx_md.end_of_burst = true;
363  first_packet_state = false;
364  last_packet_state = true;
365  } else if (flags_lsb == 4) { // start and end
366  // s->tx_md.start_of_burst = true;
367  // s->tx_md.end_of_burst = true;
368  first_packet_state = true;
369  last_packet_state = true;
370  } else if (flags_lsb==1) { // middle of burst
371  // s->tx_md.start_of_burst = false;
372  // s->tx_md.end_of_burst = false;
373  first_packet_state = false;
374  last_packet_state = false;
375  }
376  else if (flags_lsb==10) { // fail safe mode
377  // s->tx_md.has_time_spec = false;
378  // s->tx_md.start_of_burst = false;
379  // s->tx_md.end_of_burst = true;
380  first_packet_state = false;
381  last_packet_state = true;
382  }
383 
384  if(usrp_tx_thread == 0){
385 #if defined(__x86_64) || defined(__i386__)
386  #ifdef __AVX2__
387  nsamps2 = (nsamps+7)>>3;
388  __m256i buff_tx[cc<2?2:cc][nsamps2];
389  #else
390  nsamps2 = (nsamps+3)>>2;
391  __m128i buff_tx[cc<2?2:cc][nsamps2];
392  #endif
393 #elif defined(__arm__)
394  nsamps2 = (nsamps+3)>>2;
395  int16x8_t buff_tx[cc<2?2:cc][nsamps2];
396 #else
397 #error Unsupported CPU architecture, USRP device cannot be built
398 #endif
399 
400  // bring RX data into 12 LSBs for softmodem RX
401  for (int i=0; i<cc; i++) {
402  for (int j=0; j<nsamps2; j++) {
403 #if defined(__x86_64__) || defined(__i386__)
404 #ifdef __AVX2__
405  buff_tx[i][j] = _mm256_slli_epi16(((__m256i *)buff[i])[j],4);
406 #else
407  buff_tx[i][j] = _mm_slli_epi16(((__m128i *)buff[i])[j],4);
408 #endif
409 #elif defined(__arm__)
410  buff_tx[i][j] = vshlq_n_s16(((int16x8_t *)buff[i])[j],4);
411 #endif
412  }
413  }
414 
415  s->tx_md.has_time_spec = true;
416  s->tx_md.start_of_burst = (s->tx_count==0) ? true : first_packet_state;
417  s->tx_md.end_of_burst = last_packet_state;
418  s->tx_md.time_spec = uhd::time_spec_t::from_ticks(timestamp, s->sample_rate);
419  s->tx_count++;
420 
421 VCD_SIGNAL_DUMPER_DUMP_FUNCTION_BY_NAME(VCD_SIGNAL_DUMPER_FUNCTIONS_BEAM_SWITCHING_GPIO,1);
422  // bit 3 enables gpio (for backward compatibility)
423  if (flags_msb&8) {
424  // push GPIO bits 7-9 from flags_msb
425  int gpio789=(flags_msb&7)<<7;
426  s->usrp->set_command_time(s->tx_md.time_spec);
427  s->usrp->set_gpio_attr("FP0", "OUT", gpio789, 0x380);
428  s->usrp->clear_command_time();
429  }
430 VCD_SIGNAL_DUMPER_DUMP_FUNCTION_BY_NAME(VCD_SIGNAL_DUMPER_FUNCTIONS_BEAM_SWITCHING_GPIO,0);
431 
432  if (cc>1) {
433  std::vector<void *> buff_ptrs;
434 
435  for (int i=0; i<cc; i++)
436  buff_ptrs.push_back(&(((int16_t *)buff_tx[i])[0]));
437 
438  ret = (int)s->tx_stream->send(buff_ptrs, nsamps, s->tx_md);
439  }
440  else {
441  ret = (int)s->tx_stream->send(&(((int16_t *)buff_tx[0])[0]), nsamps, s->tx_md);
442  }
443 
444  if (ret != nsamps) LOG_E(HW,"[xmit] tx samples %d != %d\n",ret,nsamps);
445  return ret;
446  }
447  else{
448  pthread_mutex_lock(&write_thread->mutex_write);
449 
450  if(write_thread->count_write >= MAX_WRITE_THREAD_PACKAGE){
451  LOG_W(HW,"Buffer overflow, count_write = %d, start = %d end = %d, resetting write package\n", write_thread->count_write, write_thread->start, write_thread->end);
452  write_thread->end = write_thread->start;
453  write_thread->count_write = 0;
454  }
455 
456  end = write_thread->end;
457  write_package[end].timestamp = timestamp;
458  write_package[end].nsamps = nsamps;
459  write_package[end].cc = cc;
460  write_package[end].first_packet = first_packet_state;
461  write_package[end].last_packet = last_packet_state;
462  write_package[end].flags_msb = flags_msb;
463  for (int i = 0; i < cc; i++)
464  write_package[end].buff[i] = buff[i];
465  write_thread->count_write++;
466  write_thread->end = (write_thread->end + 1)% MAX_WRITE_THREAD_PACKAGE;
467  LOG_D(HW,"Signaling TX TS %llu\n",(unsigned long long)timestamp);
468  pthread_cond_signal(&write_thread->cond_write);
469  pthread_mutex_unlock(&write_thread->mutex_write);
470  return 0;
471  }

620 static int trx_usrp_read(openair0_device *device, openair0_timestamp *ptimestamp, void **buff, int nsamps, int cc) {
621  usrp_state_t *s = (usrp_state_t *)device->priv;
622  int samples_received=0;
623  int nsamps2; // aligned to upper 32 or 16 byte boundary
624 #if defined(__x86_64) || defined(__i386__)
625 #ifdef __AVX2__
626  nsamps2 = (nsamps+7)>>3;
627  __m256i buff_tmp[cc<2 ? 2 : cc][nsamps2];
628 #else
629  nsamps2 = (nsamps+3)>>2;
630  __m128i buff_tmp[cc<2 ? 2 : cc][nsamps2];
631 #endif
632 #elif defined(__arm__)
633  nsamps2 = (nsamps+3)>>2;
634  int16x8_t buff_tmp[cc<2 ? 2 : cc][nsamps2];
635 #endif
636 
637  int rxshift;
638  switch (device->type) {
639  case USRP_B200_DEV:
640  rxshift=4;
641  break;
642  case USRP_X300_DEV:
643  case USRP_N300_DEV:
644  rxshift=2;
645  break;
646  default:
647  AssertFatal(1==0,"Shouldn't be here\n");
648  }
649 
650  samples_received=0;
651  while (samples_received != nsamps) {
652 
653  if (cc>1) {
654  // receive multiple channels (e.g. RF A and RF B)
655  std::vector<void *> buff_ptrs;
656 
657  for (int i=0; i<cc; i++) buff_ptrs.push_back(buff_tmp[i]+samples_received);
658 
659  samples_received += s->rx_stream->recv(buff_ptrs, nsamps, s->rx_md);
660  } else {
661  // receive a single channel (e.g. from connector RF A)
662 
663  samples_received += s->rx_stream->recv((void*)((int32_t*)buff_tmp[0]+samples_received),
664  nsamps-samples_received, s->rx_md);
665  }
666  if ((s->wait_for_first_pps == 0) && (s->rx_md.error_code!=uhd::rx_metadata_t::ERROR_CODE_NONE))
667  break;
668 
669  if ((s->wait_for_first_pps == 1) && (samples_received != nsamps)) {
670  printf("sleep...\n"); //usleep(100);
671  }
672  }
673  if (samples_received == nsamps) s->wait_for_first_pps=0;
674 
675  // bring RX data into 12 LSBs for softmodem RX
676  for (int i=0; i<cc; i++) {
677  for (int j=0; j<nsamps2; j++) {
678 #if defined(__x86_64__) || defined(__i386__)
679 #ifdef __AVX2__
680  // FK: in some cases the buffer might not be 32 byte aligned, so we cannot use avx2
681 
682  if ((((uintptr_t) buff[i])&0x1F)==0) {
683  ((__m256i *)buff[i])[j] = _mm256_srai_epi16(buff_tmp[i][j],rxshift);
684  } else {
685  ((__m128i *)buff[i])[2*j] = _mm_srai_epi16(((__m128i *)buff_tmp[i])[2*j],rxshift);
686  ((__m128i *)buff[i])[2*j+1] = _mm_srai_epi16(((__m128i *)buff_tmp[i])[2*j+1],rxshift);
687  }
688 
689 #else
690  ((__m128i *)buff[i])[j] = _mm_srai_epi16(buff_tmp[i][j],rxshift);
691 #endif
692 #elif defined(__arm__)
693  ((int16x8_t *)buff[i])[j] = vshrq_n_s16(buff_tmp[i][j],rxshift);
694 #endif
695  }
696  }
697 
698  if (samples_received < nsamps) {
699  LOG_E(HW,"[recv] received %d samples out of %d\n",samples_received,nsamps);
700  }
701 
702  if ( s->rx_md.error_code != uhd::rx_metadata_t::ERROR_CODE_NONE)
703  LOG_E(HW, "%s\n", s->rx_md.to_pp_string(true).c_str());
704 
705  s->rx_count += nsamps;
706  s->rx_timestamp = s->rx_md.time_spec.to_ticks(s->sample_rate);
707  *ptimestamp = s->rx_timestamp;
708 
709  // push GPIO bits 7-9 from flags_msb
710  /*s->usrp->set_command_time(uhd::time_spec_t::from_ticks((s->rx_timestamp+(2*nsamps)),s->sample_rate));
711  s->usrp->set_gpio_attr("FP0", "OUT", gpio789<<7, 0x380);
712  s->usrp->clear_command_time();
713  gpio789 = (gpio789+1)&7;*/
714  recplay_state_t *recPlay=device->recplay_state;
715 
716  if ( recPlay != NULL) { // record mode
717  // Copy subframes to memory (later dump on a file)
718  if (recPlay->nbSamplesBlocks < device->openair0_cfg->recplay_conf->u_sf_max &&
719  recPlay->maxSizeBytes > (recPlay->currentPtr-(uint8_t *)recPlay->ms_sample) +
720  sizeof(iqrec_t) + nsamps*4 ) {
721  iqrec_t *hdr=(iqrec_t *)recPlay->currentPtr;
722  hdr->header = BELL_LABS_IQ_HEADER;
723  hdr->ts = *ptimestamp;
724  hdr->nbBytes=nsamps*4;
725  memcpy(hdr+1, buff[0], nsamps*4);
726  recPlay->currentPtr+=sizeof(iqrec_t)+nsamps*4;
727  recPlay->nbSamplesBlocks++;
728  LOG_D(HW,"recorded %d samples, for TS %lu, shift in buffer %ld\n", nsamps, hdr->ts, recPlay->currentPtr-(uint8_t *)recPlay->ms_sample);
729  } else
730  exit_function(__FILE__, __FUNCTION__, __LINE__,"Recording reaches max iq limit\n");
731  }
732 
733  return samples_received;
734 }
