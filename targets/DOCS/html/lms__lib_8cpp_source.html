119 void set_rx_gain_offset(openair0_config_t *openair0_cfg, int chain_index) {
120 
121  int i=0;
122  // loop through calibration table to find best adjustment factor for RX frequency
123  double min_diff = 6e9,diff;
124 
125  while (openair0_cfg->rx_gain_calib_table[i].freq>0) {
126  diff = fabs(openair0_cfg->rx_freq[chain_index] - openair0_cfg->rx_gain_calib_table[i].freq);
127  printf("cal %d: freq %f, offset %f, diff %f\n",
128  i,
129  openair0_cfg->rx_gain_calib_table[i].freq,
130  openair0_cfg->rx_gain_calib_table[i].offset,diff);
131  if (min_diff > diff) {
132  min_diff = diff;
133  openair0_cfg->rx_gain_offset[chain_index] = openair0_cfg->rx_gain_calib_table[i].offset;
134  }
135  i++;
136  }
137 
138 }

163 int trx_lms_start(openair0_device *device){
164 
165  lms_info_str_t list[16]={0};
166 
167  int n= LMS_GetDeviceList(list);
168 
169  if (n <= 0) {
170  fprintf(stderr, "No LimeSDR board found: %s\n", n < 0?LMS_GetLastErrorMessage():"");
171  return -1;
172  }
173 
174  printf("Connecting to device: %s\n",list[0]);
175  if (LMS_Open(&lms_device,list[0],NULL)<0) {
176  fprintf(stderr, "Can't open device port: %s\n",LMS_GetLastErrorMessage());
177  return -1;
178  }
179 
180  LMS_Init(lms_device);
181  LMS_EnableCache(lms_device,false);
182 
183  if (LMS_LoadConfig(lms_device,device->openair0_cfg[0].configFilename) != 0)
184  {
185  printf("Failed to load configuration file %s\n%s",device->openair0_cfg[0].configFilename,LMS_GetLastErrorMessage());
186  exit(-1);
187  }
188 
189  /* LMS_EnableChannel(lms_device,LMS_CH_RX,0,true);
190  LMS_EnableChannel(lms_device,LMS_CH_TX,0,true);
191  if (device->openair0_cfg->rx_num_channels == 2)
192  {
193  LMS_EnableChannel(lms_device,LMS_CH_RX,1,true);
194  LMS_EnableChannel(lms_device,LMS_CH_TX,1,true);
195  } */
196  LMS_VCTCXOWrite(lms_device, 129);
197 
198  if (LMS_SetSampleRate(lms_device,device->openair0_cfg->sample_rate,2)!=0)
199  {
200  fprintf(stderr, "Failed to set sample rate %s\n",LMS_GetLastErrorMessage());
201  return -1;
202  }
203  printf("Set sample rate %f MHz\n",device->openair0_cfg->sample_rate/1e6);
204 
205  if (LMS_SetLOFrequency(lms_device,LMS_CH_RX, 0, device->openair0_cfg[0].rx_freq[0])!=0)
206  {
207  fprintf(stderr, "Failed to Set Rx frequency: %s\n", LMS_GetLastErrorMessage());
208  return -1;
209  }
210  if (LMS_SetLOFrequency(lms_device,LMS_CH_TX, 0,device->openair0_cfg[0].tx_freq[0])!=0)
211  {
212  fprintf(stderr, "Failed to Set Tx frequency: %s\n", LMS_GetLastErrorMessage());
213  return -1;
214  }
215  printf("Set TX frequency %f MHz\n",device->openair0_cfg[0].tx_freq[0]/1e6);
216 
217  /*
218  printf("Override antenna settings to: RX1_H, TXA_2");
219  LMS_SetAntenna(lms_device, LMS_CH_RX, 0, 1);
220  LMS_SetAntenna(lms_device, LMS_CH_TX, 0, 2);
221  */
222 
223 
224  for (int i = 0; i< device->openair0_cfg->rx_num_channels; i++)
225  {
226  if (LMS_SetLPFBW(lms_device,LMS_CH_RX,i,device->openair0_cfg->rx_bw)!=0)
227  printf("RX ch:%d filter calibration failed, bw:%fMHz\n%s\n",i,device->openair0_cfg->rx_bw/1e6,LMS_GetLastErrorMessage());
228  if (LMS_SetLPFBW(lms_device,LMS_CH_TX,i,device->openair0_cfg->tx_bw)!=0)
229  printf("TX ch:%d filter calibration failed, bw:%fMHz\n%s\n",i,device->openair0_cfg->tx_bw/1e6,LMS_GetLastErrorMessage());
230 
231  if (LMS_Calibrate(lms_device,LMS_CH_RX,i,device->openair0_cfg->rx_bw,0)!=0)
232  printf("RX ch:%d calibration failed, bw:%f MHz\n%s\n",i,device->openair0_cfg->rx_bw/1e6,LMS_GetLastErrorMessage());
233  if (LMS_Calibrate(lms_device,LMS_CH_TX,i,device->openair0_cfg->tx_bw,0)!=0)
234  printf("TX ch:%d calibration failed, bw:%fMHz\n%s\n",i,device->openair0_cfg->tx_bw/1e6,LMS_GetLastErrorMessage());
235  }
236 
237 
238  first_rx = 1;
239 
240  rx_stream.channel = 0;
241  rx_stream.fifoSize = 256*1024;
242  rx_stream.throughputVsLatency = 0.1;
243  rx_stream.dataFmt = lms_stream_t::LMS_FMT_I12;
244  rx_stream.isTx = false;
245  if (LMS_SetupStream(lms_device, &rx_stream)!=0)
246  printf("RX stream setup failed %s\n",LMS_GetLastErrorMessage());
247  tx_stream.channel = 0;
248  tx_stream.fifoSize = 256*1024;
249  tx_stream.throughputVsLatency = 0.1;
250  tx_stream.dataFmt = lms_stream_t::LMS_FMT_I12;
251  tx_stream.isTx = true;
252 
253  trx_lms_set_gains(device, device->openair0_cfg);
254 
255  if (LMS_SetupStream(lms_device, &tx_stream)!=0)
256  printf("TX stream setup failed %s\n",LMS_GetLastErrorMessage());
257 
258  printf("SR: %.3f MHz\n", (float)device->openair0_cfg->sample_rate / 1e6);
259 
260 
261  printf("SR: %.3f MHz\n", (float)device->openair0_cfg->sample_rate / 1e6);
262 
263  if (LMS_StartStream(&rx_stream)!=0)
264  printf("Failed to start TX stream %s\n",LMS_GetLastErrorMessage());
265  if (LMS_StartStream(&tx_stream)!=0)
266  printf("Failed to start Rx stream %s\n",LMS_GetLastErrorMessage());
267  return 0;
268 }
